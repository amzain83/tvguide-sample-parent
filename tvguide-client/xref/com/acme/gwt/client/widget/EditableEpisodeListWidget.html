33  /**
34   * Facilitates editing episode info inline, not requiring any list, focus, edit, save, list cycle.
35   * Also shows off the CellTable-Tools, for better or worse.
36   * 
37   * 
38   * @author colin
39   */
40  public class EditableEpisodeListWidget extends Composite
41  		implements
42  			IsEditor<HasDataEditor<TvEpisodeProxy>> {
43  	private static Binder uiBinder = GWT.create(Binder.class);
44  
45  	interface Binder extends UiBinder<Widget, EditableEpisodeListWidget> {
46  	}
47  
48  	interface EpisodeColumns extends Columns<TvEpisodeProxy> {
49  		@Editable
50  		@ConvertedWith(IntegerConverter.class)
51  		EditTextCell season();//erp int != String
52  
53  		@Editable
54  		@ConvertedWith(IntegerConverter.class)
55  		EditTextCell episodeNumber();// int != String
56  
57  		@Editable
58  		EditTextCell name();
59  	}
60  
61  	private EpisodeColumns columns = GWT.create(EpisodeColumns.class);
62  	@Path("")
63  	HasDataEditor<TvEpisodeProxy> listEd;
64  	@UiField(provided = true)
65  	CellTable<TvEpisodeProxy> list = new CellTable<TvEpisodeProxy>();
66  
67  	public EditableEpisodeListWidget() {
68  		listEd = HasDataEditor.of(list);
69  
70  		columns.configure(list);
71  
72  		initWidget(uiBinder.createAndBindUi(this));
73  	}
74  
75  	@Override
76  	public HasDataEditor<TvEpisodeProxy> asEditor() {
77  		return listEd;
78  	}
79  }
